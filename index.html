Make a Backend server

Step by Step

1-npm init -y

2-make a entry point - make a file and name it app.js

3-npm install express

4-get this code in app.js 


      // get this code in app.js
      const express = require('express')
      const app = express()
      const port = 3000
      
      app.get('/', (req, res) => {
        res.send('Hello World!')
      })
      
      app.listen(port, () => {
        console.log(`http://localhost:${port}`);
      })

5-npm install -g nodemon

6-now you can run your server from tapping in the terminal (nodemon app.js)

// nodemon responsable to read all changes in our file

// that's mean the server going to refresh every time we make changes

7-now we must make a folder named views and make file named home.html inside it

8-in app.js let's change res.send('Hello World!') with home file we did create 


      // get this code in app.js
      const express = require('express')
      const app = express()
      const port = 3000
      
      app.get('/', (req, res) => {
        res.sendFile("./views/home.html", {root: __dirname});
      })
      
      app.listen(port, () => {
        console.log(`http://localhost:${port}`);
      })
